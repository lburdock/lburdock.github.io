Duplicating tests#

test.each([
{ a: 1, b: 1, expected: 2 },
{ a: 1, b: 2, expected: 3 },
{ a: 2, b: 1, expected: 3 },
])("$a + $b === $expected", ({ a, b, expected }) => {
expect(a + b).toBe(expected);
});
test.each`
a | b | expected
${1} | ${1} | ${2}
${1} | ${2} | ${3}
${2} | ${1} | ${3}
`("$a + $b === $expected", ({ a, b, expected }) => {
expect(a + b).toBe(expected);
});

Duplicating test suites#

describe.each([
{ a: 1, b: 1, expected: 2 },
{ a: 1, b: 2, expected: 3 },
{ a: 2, b: 1, expected: 3 },

expect(a + b).not.toBeGreaterThan(expected);
});

expect(a + b).not.toBeLessThan(expected);
});
});

React Test Library (+ Jest)#

Test syntax#

button.test.jsx
import React from "react";
import { render, screen } from "@testing-library/react";
import Button from "./button";
describe("Button", () => {
test("renders", () => {
render(<Button>Okay</Button>);
const btn = screen.getByRole("button");
expect(btn).toBeInTheDocument();
});
});

User event example#

import React from "react";
import { render, screen } from "@testing-library/react";
import userEvent from "@testing-library/user-event";
import Button from "./button";
describe("Button", () => {
test("renders", () => {
const onClick = jest.fn();
render(<Button onClick={onClick}>Okay</Button>);
userEvent.click(screen.getByRole("button"));
expect(onClick).toHaveBeenCalled();
});
});

Async testing#

import {
render,
screen,
waitFor,
waitForElementToBeRemoved,
} from "@testing-library/react";
test("renders", () => {
render(<Button>Okay</Button>);
// Preferred: Use `findBy` to wait for the element to appear
expect(await screen.findByRole("button")).toBeInTheDocument();
// `findBy` is equivalent to using `waitFor` with `getBy`
await waitFor(() =>
expect(screen.getByRole("button")).toBeInTheDocument()
);
// Shorthand for element removal
await waitForElementToBeRemoved(screen.getByRole("button"));
});
